\section{Problem Definition}
%% ---------------------------------------------------------------

\begin{definition}[Budgeted Connected Case Segmentation --- BCCS]
\label{def:bccs}
Let $G=(V,E)$ be an undirected graph, $L\subseteq E$ a set of
\emph{target edges}, and $B\in\mathbb{Z}^+$ a \emph{budget}.
A \textbf{case assignment} is a partition
$\mathcal{P}=\{C_1,\dots,C_m\}$ of~$V$ into pairwise-disjoint,
non-empty subsets called \emph{cases}.

For each case~$C_i$ define the \textbf{induced edge set}
\[
  E(C_i)=\bigl\{(u,v)\in E : u\in C_i \text{ and } v\in C_i\bigr\}.
\]
The \textbf{coverage} of~$\mathcal{P}$ is
\[
  \operatorname{Cov}(\mathcal{P})
  =\bigl|\{(u,v)\in L:\exists\, C_i\in\mathcal{P}
            \text{ with } u,v\in C_i\}\bigr|.
\]
The \textbf{BCCS optimisation problem} is
\[
  \max_{\mathcal{P}}\;\operatorname{Cov}(\mathcal{P})
  \qquad\text{s.t.}\quad |E(C_i)|\le B
  \;\;\forall\, C_i\in\mathcal{P}.
\]
The \textbf{BCCS decision problem} asks: given $(G,L,B,\tau)$, does
there exist a partition~$\mathcal{P}$ with
$|E(C_i)|\le B$ for all~$i$ and
$\operatorname{Cov}(\mathcal{P})\ge\tau$?
\end{definition}

\paragraph{Interpretation in AML transaction monitoring.}
In anti-money laundering (AML) systems, $G$~represents the full
transaction graph, $L$~is the set of top-$k$ highest-risk edges
identified by a GNN scorer, and $B$~limits the number of edges an
analyst can review in a single investigation case.  BCCS asks how to
group transactions into bounded investigation cases to maximise the
number of flagged edges that are actually reviewable.

%% ---------------------------------------------------------------
\section{NP-Completeness}
%% ---------------------------------------------------------------

\begin{theorem}\label{thm:npc}
BCCS is NP-complete.
\end{theorem}

\begin{proof}

\subsection*{Membership in NP}

Given a partition $\mathcal{P}=\{C_1,\dots,C_m\}$, we can verify in
$O(|V|+|E|)$ time that (i)~$|E(C_i)|\le B$ for every~$i$
and (ii)~$\operatorname{Cov}(\mathcal{P})\ge\tau$, by building a
node-to-case lookup table and scanning all edges once.
Hence $\text{BCCS}\in\text{NP}$.

\subsection*{NP-Hardness (reduction from Maximum Independent Set)}

We reduce from the \textsc{Maximum Independent Set} problem (MIS),
known to be NP-hard~\cite{karp1972}.

\medskip\noindent
\textbf{MIS.}\;
Given a graph $H=(V_H,E_H)$ and an integer~$k$, does~$H$ contain an
independent set of size~$\ge k$?

\medskip\noindent
\textbf{Construction.}\;
Given a MIS instance $(H,k)$, build the BCCS instance
$(G,L,B,\tau)$ as follows.

\begin{enumerate}
\item \emph{Add a goal node.}
      Let~$g$ be a fresh node; set $V=V_H\cup\{g\}$.

\item \emph{Edges.}
      $E = E_H \;\cup\; \{(v,g) : v\in V_H\}$.

\item \emph{Target edges.}
      $L = \{(v,g) : v\in V_H\}$.
      Only the edges to~$g$ are targets.

\item \emph{Budget and threshold.}
      $B = \tau = k$.
\end{enumerate}

The construction runs in $O(|V_H|+|E_H|)$ time.

\medskip\noindent
\textbf{Intuition.}\;
Covering a target edge $(v,g)$ requires~$v$ and~$g$ in the same case.
Because~$g$ belongs to a single case, all ``covered'' nodes share that
case with~$g$.  The non-target edges from~$E_H$ between those nodes
consume budget, so only independent sets fit within budget~$k$.

\medskip\noindent
\textbf{Forward direction ($\Rightarrow$).}\;
Suppose~$H$ has an independent set $S\subseteq V_H$ with $|S|=k$.
Define the partition $C_1=S\cup\{g\}$; every remaining node forms a
singleton case.

\begin{itemize}
\item \emph{Budget of~$C_1$:}\;
      $E(C_1)=E_H(S)\cup\{(v,g):v\in S\}$.
      Since~$S$ is independent, $E_H(S)=\emptyset$, so
      $|E(C_1)|=|S|=k=B$.\;\checkmark

\item \emph{Singletons:}\; $0$~edges each.\;\checkmark

\item \emph{Coverage:}\;
      Each $v\in S$ is co-located with~$g$, giving
      $\operatorname{Cov}=k=\tau$.\;\checkmark
\end{itemize}

\medskip\noindent
\textbf{Backward direction ($\Leftarrow$).}\;
Suppose a partition~$\mathcal{P}$ achieves
$\operatorname{Cov}(\mathcal{P})\ge k$.
Let~$C_g$ be the case containing~$g$ and set
$S=C_g\cap V_H$.

A target edge $(v,g)$ is covered if and only if $v\in S$, so
$\operatorname{Cov}(\mathcal{P})=|S|\ge k$.

The induced edges of~$C_g$ are
\[
  E(C_g) = E_H(S)\;\cup\;\{(v,g):v\in S\},
\]
hence $|E(C_g)|=|E_H(S)|+|S|$.
The budget constraint gives
\[
  |E_H(S)|+|S|\;\le\; B = k.
\]
Since $|S|\ge k$, we obtain $|E_H(S)|\le k-|S|\le 0$.
Therefore $|E_H(S)|=0$ and $|S|=k$:
the set~$S$ is an independent set of size~$k$ in~$H$.\;\checkmark

\medskip
This completes the reduction.
\end{proof}

%% ---------------------------------------------------------------
\section{Consequences}
%% ---------------------------------------------------------------

\subsection{Inapproximability}

The reduction preserves the objective value exactly
($\operatorname{Cov}=|S|$).  Any $\alpha$-approximation for BCCS
therefore yields an $\alpha$-approximation for MIS.
Zuckerman~\cite{zuckerman2007} showed that MIS cannot be approximated
within $|V|^{1-\varepsilon}$ for any $\varepsilon>0$ unless P\,=\,NP.

\begin{corollary}\label{cor:inapprox}
Unless $\mathrm{P}=\mathrm{NP}$, BCCS cannot be approximated within
$|V|^{1-\varepsilon}$ for any $\varepsilon>0$.
\end{corollary}

This worst-case bound applies to adversarial instances.
Real-world AML graphs exhibit temporal locality and natural community
structure that make the problem far more tractable in practice.

\subsection{Connection to Maximum Coverage}

BCCS also relates to the \textsc{Maximum Coverage Problem} (MCP).
In settings where each case acts as a ``set'' of edges it covers,
the BCCS optimiser must decide which target edges to cover within
the budget, analogous to selecting sets in MCP.
Feige~\cite{feige1998} showed that MCP cannot be approximated beyond
$1-1/e\approx 0.632$ unless P\,=\,NP.

\subsection{Practical Implications}

Despite worst-case NP-hardness, our hierarchical WCC\,+\,Leiden
heuristic (BTCS) achieves \textbf{94--97\%} of the upper bound
(NoBudget WCC coverage) at $k=1\%$ on the AML100k and AML1M datasets.
This suggests that real-world AML transaction graphs possess structure
--- e.g.\ natural community boundaries that align with the budget ---
making the problem tractable in practice even though worst-case
instances are intractable.

\section{Approximation Guarantees for BTCS}
%% ---------------------------------------------------------------

Although BCCS is NP-hard in general, we establish four formal results
that characterise the performance of the BTCS algorithm (hierarchical
WCC\,+\,Leiden) on structured instances.  Together, these results
explain the strong empirical performance observed across 15 benchmark
datasets.

%%----
\subsection{WCC Coverage as a Computable Upper Bound}

Let $G_k=(V_k,E_k)$ denote the subgraph of $G$ induced by the
top-$k$ edges (the \emph{top-$k$ subgraph}).  Define:
\[
  \text{WCC-COV}(G,G_k,L)
  =\bigl|\{(u,v)\in L : u\text{ and }v
         \text{ are in the same WCC of }G_k\}\bigr|.
\]

\begin{theorem}[WCC Coverage Upper Bound for BTCS]\label{thm:wcc-ub}
For all graphs $G$, top-$k$ subgraphs $G_k$, target sets $L$,
and budgets $B\ge 1$:
\[
  \mathrm{BTCS}(G,G_k,L,B)\;\le\;\mathrm{WCC\text{-}COV}(G,G_k,L).
\]
\end{theorem}

\begin{proof}
BTCS partitions nodes according to $\mathrm{WCC}(G_k)$: two nodes
$u,v$ are placed in the same case only if they belong to the same WCC
(or a Leiden sub-partition thereof).  Hence a target edge $(u,v)\in L$
can be covered only if $u$ and~$v$ are already co-located in the same
WCC of~$G_k$.  Therefore $\mathrm{BTCS}\le\mathrm{WCC\text{-}COV}$.
\end{proof}

\begin{remark}
WCC-COV is computable in $O(|V|+|E_k|)$ and is strictly tighter than
$|L|$.  Empirically it equals 91.2\% and 90.1\% of $|L|$ on AML100k
and AML1M at $k=1\%$, so at most 8--9\% of target edges are
unreachable even without a budget constraint.
\end{remark}

%%----
\subsection{BTCS Dominates Budget-Capped WCC}

The \emph{budget-capped WCC baseline} (WCC-CAP): (i)~compute WCC,
(ii)~treat each component as one case, (iii)~apply score-descending
truncation to $B$ induced edges.  For a WCC component $C$ with
$m=|E_G(C)|$ induced edges, $\ell=|L\cap E_G(C)|$ target edges, and
$q(C)=\lceil m/B\rceil$:

\begin{theorem}[BTCS Multiplicative Gain over WCC-CAP]\label{thm:gain}
Let $f_{\mathrm{cut}}(C)=|L_{\mathrm{cut}}(C)|/\ell$ be the fraction
of target edges in~$C$ cut between Leiden sub-cases.  Then:
\[
  \mathrm{cov}_{\mathrm{BTCS}}(C)
  \;\ge\;\ell\bigl(1-f_{\mathrm{cut}}(C)\bigr)
  \;\ge\;\mathrm{cov}_{\mathrm{CAP}}(C)\cdot q(C)
         \cdot\bigl(1-f_{\mathrm{cut}}(C)\bigr).
\]
When $f_{\mathrm{cut}}(C)\le 1-1/q(C)$, BTCS covers at least $q(C)$
times more target edges than WCC-CAP in component~$C$.
\end{theorem}

\begin{proof}
A target edge $(u,v)\in L\cap E_G(C)$ is covered by BTCS iff $u,v$
land in the same Leiden sub-case, so
$\mathrm{cov}_{\mathrm{BTCS}}(C)=\ell(1-f_{\mathrm{cut}})$.
WCC-CAP retains~$B$ of~$m$ induced edges, covering at most
$\ell\cdot B/m=\ell/q(C)$ target edges.  Dividing:
$\mathrm{cov}_{\mathrm{BTCS}}/\mathrm{cov}_{\mathrm{CAP}}
\ge q(C)(1-f_{\mathrm{cut}})$.
\end{proof}

\paragraph{Empirical validation.}
At $k=5\%$ on AML1M, large WCC components have $m\approx3{,}000$--$20{,}000$
and $B=100$, giving $q(C)\approx30$--$200$.  Observed:
BTCS\,=\,0.621 vs.\ WCC-CAP\,=\,0.138, a factor of $4.5\times$,
consistent with Theorem~\ref{thm:gain}.

%%----
\subsection{Coverage Gap Characterisation}

\begin{definition}[Cross-WCC Target Edges]
\[
  \mathrm{cross}(G,G_k,L)
  =\bigl|\{(u,v)\in L : u\text{ and }v\text{ are in \emph{different} WCCs of }G_k\}\bigr|.
\]
\end{definition}

\begin{theorem}[Coverage Gap of BTCS vs.\ OPT]\label{thm:gap}
\[
  \mathrm{OPT}(G,L,B)-\mathrm{BTCS}(G,G_k,L,B)
  \;\le\;\mathrm{cross}(G,G_k,L)+|L_{\mathrm{cut}}|,
\]
where $L_{\mathrm{cut}}=\sum_{C\;\text{over-budget}}L_{\mathrm{cut}}(C)$
are target edges cut by Leiden within over-budget WCC components.
\end{theorem}

\begin{proof}
Target edges covered by OPT but not BTCS fall into two disjoint groups:
(i)~cross-WCC edges: OPT may merge nodes from different WCCs; BTCS cannot,
    so at most $\mathrm{cross}(G,G_k,L)$ such edges are missed;
(ii)~intra-WCC Leiden cuts: for over-budget~$C$, Leiden may separate
    $(u,v)\in L\cap E_G(C)$; these total $|L_{\mathrm{cut}}|$.
The gap is bounded by their sum.
\end{proof}

\begin{corollary}[Instance-Specific Approximation Ratio]\label{cor:approx}
Let $\delta=(\mathrm{cross}+|L_{\mathrm{cut}}|)/|L|$.  Then:
\[
  \mathrm{BTCS}(G,G_k,L,B)\;\ge\;(1-\delta)\cdot\mathrm{OPT}(G,L,B).
\]
In AML graphs with strong temporal locality, $\delta$ is empirically
small, yielding a near-optimal instance-specific ratio.
\end{corollary}

%%----
\subsection{Budget-Feasibility Rate and Near-Optimality}

\begin{definition}[Budget-Feasible Instance]
Instance $(G,G_k,L,B)$ is \emph{budget-feasible} if every WCC component
$C$ of $G_k$ satisfies $|E_G(C)|\le B$ (no Leiden splitting required).
\end{definition}

\begin{theorem}[BTCS Optimality on Budget-Feasible Instances]\label{thm:bfi}
On budget-feasible instances:
\[
  \mathrm{BTCS}(G,G_k,L,B)=\mathrm{WCC\text{-}COV}(G,G_k,L),
  \qquad L_{\mathrm{cut}}=\emptyset.
\]
The gap vs.\ OPT reduces to $\mathrm{cross}(G,G_k,L)$ alone.
\end{theorem}

\begin{proof}
When all WCCs satisfy $|E_G(C)|\le B$, BTCS retains each WCC as a
single case (no Leiden splitting), covering all intra-WCC target edges.
This equals WCC-COV by definition; no intra-WCC cuts occur.
\end{proof}

\begin{proposition}[Budget-Feasibility Rate in AML Graphs]
At $k=1\%$ with $B=100$, the AML100k dataset has 2,546 cases of which
only 38 require splitting (BFR\,=\,98.5\%), and AML1M has 22,170 cases
of which 448 require splitting (BFR\,=\,98.0\%).
At $k=1\%$, BTCS operates in the near-budget-feasible regime,
achieving 96.5\% and 94.2\% of the NoBudget upper bound,
respectively. The remaining gap comes entirely from the $\approx2\%$
of cases requiring Leiden splitting.
\end{proposition}

\begin{remark}[Coverage Degradation at $k=5\%$]
As~$k$ increases, larger WCC components form: at $k=5\%$ on AML1M,
over 99\% of top-$k$ edges belong to components exceeding $B=100$.
Budget-feasibility collapses and WCC-CAP degrades to 13.8\% coverage.
BTCS's Leiden splitting recovers 62.1\% via the multiplicative gain of
Theorem~\ref{thm:gain} ($q(C)\gg1$ for large components),
confirming the $4.5\times$ empirical gain at $k=5\%$.
\end{remark}
